## Example

export const SendTransaction = () => {
  const [account, accountState] = useAuth()
  const [transactions, transactionState] = useTransaction(
    account,
    accountState.chainId
  )
  const [sending, setSending] = useState(false)
  useEffect(() => {
    if (sending) {
      Promise.resolve()
        .then(() =>
          new Web3Provider(accountState.provider).getSigner().sendTransaction({
            from: account,
            to: account,
            value: 1,
          })
        )
        .then((tx) => {
          transactionState.add(tx.hash)
          setSending(false)
        })
        .catch((err) => {
          console.error(err)
          setSending(false)
        })
    }
  }, [sending])
  return (
    <Container>
      <Grid>
        {!account && (
          <Grid.Row>
            <Grid.Column tablet="16">
              <Blockquote>
                <Paragraph small secondary>
                  This feature requires to login, try to connect using the{' '}
                  <Link href="./?path=/docs/hooks-useauth--docs#connect-preview">
                    `useAuth`
                  </Link>{' '}
                  hook.
                </Paragraph>
              </Blockquote>
            </Grid.Column>
          </Grid.Row>
        )}
        <Grid.Row>
          <Grid.Column tablet="16">
            <Button
              primary
              size="small"
              style={{ minWidth: '200px', margin: '0 1em 0 0' }}
              disabled={!account}
              loading={sending}
              onClick={() => setSending(true)}
            >
              Send transaction
            </Button>
            <Button
              primary
              size="small"
              style={{ minWidth: '200px', margin: '0 1em 0 0' }}
              disabled={!account || transactions.length === 0}
              loading={sending}
              onClick={() => transactionState.clear()}
            >
              Clear
            </Button>
            <Source
              language="ts"
              code={JSON.stringify(transactions, null, 2)}
            />
          </Grid.Column>
        </Grid.Row>
      </Grid>
    </Container>
  )
}

```tsx dark=true
import React from 'react'
import { Web3Provider } from '@ethersproject/providers'

export function Sign() {
  const [address, state] = useAuth()
  const [transactions, transactionsState] = useTransaction(
    account,
    state.chainId
  )

  function handleTransaction() {
    new Web3Provider(state.provider)
      .getSigner()
      .sendTransaction({
        from: account,
        to: account,
        value: 1,
      })
      .then((tx) => transactionsState.add(tx.hash))
  }

  return (
    <div>
      <button disabled={!account} onClick={handleTransaction}>
        send transaction
      </button>
      <pre>{JSON.stringify(transactions, null, 2)}</pre>
    </div>
  )
}
```

## Related hooks

- [`useAuth`](./?path=/docs/hooks-useauth--docs): allows to create a user session using one of the availables providers
- [`useSign`](./?path=/docs/hooks-usesign--docs): allow to sign a message using a pair address and provider
- [`useTransaction`](./?path=/docs/hooks-usetransaction--docs): allow to track transactions
- [`useAuthContext`](./?path=/docs/context-auth-useauthcontext--docs): use a single instance of `useAuth` injected through a provider
- [`useSignContext`](./?path=/docs/context-auth-usesigncontext--docs): same as `useSign` but detects params automatically usgin the `useAuthContext` hook
- [`useTransactionContext`](./?path=/docs/context-auth-usetransactioncontext--docs): same as `useTransaction` but detects params automatically usign the `useAuthContext` hook
