# Clipboard Copy Hook

> The *clipboard copy hook* allows to copy an arbitrary string to the user clipboards.

```ts
const [ value, clipboard ] = useClipboardCopy(timeout)
```

<ArgsTable rows={[
  Args.param('timeout', 'time, in milliseconds, when the value will be reset to `null`', 'number'),
  Args.returns('value', 'latest value copied using `clipboard.copy`', `boolean | number | string | null`),
  Args.returns('clipboard.copy', 'send value to the user clipboard', `(value: boolean | number | string) => void`),
  Args.returns('clipboard.clear', 'reset clipboard value to `null`', `() => void`),
]} />

## Preview without clipboard

export const WithoutTimeout = () => {
  const [ value, clipboard ] = useClipboardCopy()
  return <Container>
    <Button primary size="small" onClick={() => clipboard.copy(randomHexadecimal(24))}>
      copy a random
    </Button>
    <Button primary size="small" onClick={() => clipboard.clear()}>
      clear
    </Button>
    <Source language="json" code={JSON.stringify(value, null, 2)} />
  </Container>
}

```tsx dark=true
import React from "react"
import Time from 'decentraland-gatsby/dist/utils/date/Time'
import { randomHexadecimal } from 'decentraland-gatsby/dist/utils/string/random'

export function Example () {
  const [ value, clipboard ] = useClipboardCopy(3 * Time.Second)

  return <div>
    <button onClick={() => clipboard.copy(randomHexadecimal())}>
      copy random with timeout
    </button>
    <pre>{JSON.stringify(value, null, 2)}</pre>
  </div>
}
```
