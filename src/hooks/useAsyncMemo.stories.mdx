# Auth Hook

> The *async memo* allows to resolve and async function and stores its result in the component livecycle.

```ts
const [ value, state ] = useAsyncMemo(resolver, deps, options)
```

<ArgsTable rows={[
  Args.param('resolver', 'async resolver', `() => Promise<T>`),
  Args.param('deps', 'dependencies', `React.DependencyList`),
  Args.param('options.initialValue', 'initail value', `any`),
  Args.param('options.callWithTruthyDeps', 'if `true` resolver will be call only when all the dependencies are evaluated as `true`', `boolean`),
  Args.returns('value', 'value returned by the resolver', `T || options.initialValue || null`),
  Args.returns('state.loading', '`true` when is waiting to be resolved', `boolean`),
  Args.returns('state.version', '`true` when is waiting to be resolved', `boolean`),
  Args.returns('state.time', 'the time (in milliseconds) that takes to call the resolver', `number`),
  Args.returns('state.error', 'contains the error for the latest resolver call', `Error | null`),
  Args.returns('state.reload', 're-execute the call to the resolver', `() => void`),
  Args.returns('state.set', 'set an arbitrary value', `(value: T) => void`),
]} />

export const Example = () => {
  const [ value, state ] = useAsyncMemo(() => fetch('https://httpbin.org/get').then(r => r.json()), [], {})
  return <Container>
    <Button primary small onClick={() => state.reload()}>reload</Button>
    <Button primary small onClick={() => state.set(random(1000, 9999))}>set random</Button>
    <Source language="ts" code={'// value \n\n' + JSON.stringify(value, null, 2)} />
    <Source language="ts" code={'// state \n\n' +JSON.stringify({ ...state, error: toObject(state.error) }, null, 2)} />
  </Container>
}

export const ErrorExample = () => {
  const [ value, state ] = useAsyncMemo(async () => {
    const time = random(1000, 3000)
    await delay(time)
    throw new Error(`[Error: ${time}] random error`)
  }, [], {})
  return <Container>
    <Button primary small onClick={() => state.reload()}>reload</Button>
    <Button primary small onClick={() => state.set(random(1000, 9999))}>set random</Button>
    <Source language="ts" code={'// value \n\n' + JSON.stringify(value, null, 2)} />
    <Source language="ts" code={'// state \n\n' +JSON.stringify({ ...state, error: toObject(state.error) }, null, 2)} />
  </Container>
}
