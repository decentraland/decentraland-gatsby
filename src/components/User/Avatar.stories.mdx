# Avatar component

> Render the user faces using their wallet address

export const PreviewAvatar = () => {
  const [auth] = useAuth()
  const [state, setState] = useState({
    address: '0x05d48ee3e815bf376fc79d283301cfdef872e280',
    size: Size.Full,
    src: '',
  })
  useEffect(() => {
    auth && setState((current) => ({ ...current, address: auth }))
  }, [auth])
  return (
    <Container>
      <div style={{ display: 'flex' }}>
        <div
          style={{
            flex: '0 0 40%',
            display: 'flex',
            justifyContent: 'center',
            alignItems: 'center',
          }}
        >
          <Avatar src={state.src} address={state.address} size={state.size} />
        </div>
        <div style={{ flex: '0 0 60%' }}>
          <ArgsTable
            compact={true}
            updateArgs={(value) =>
              setState((current) => ({ ...current, ...value }))
            }
            rows={{
              address: {
                name: 'address',
                control: { type: 'text', value: state.address },
              },
              size: {
                name: 'size',
                control: {
                  type: 'select',
                  value: state.size,
                  options: [
                    Size.Mini,
                    Size.Tiny,
                    Size.Small,
                    Size.Medium,
                    Size.Large,
                    Size.Big,
                    Size.Huge,
                    Size.Massive,
                    Size.Full,
                  ],
                },
              },
              src: {
                name: 'src',
                control: {
                  type: 'text',
                  value: state.src,
                  placeholder: 'https://....',
                },
              },
            }}
          />
        </div>
      </div>
      <Source
        language="ts"
        dark
        code={`<Address${[
          state.src ? ` src="${state.src}"` : ``,
          state.address ? ` address="${state.address}"` : ``,
          state.size ? ` size="${state.size}"` : ``,
        ].join('')}/>`}
      />
    </Container>
  )
}
