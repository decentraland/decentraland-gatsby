export function AuthChainCURL({ chain, timestamp, metadata, options }) {
  let curl = []
  if (options.url) {
    curl.push(`curl '${options.url}'`)
  }
  if (options.method) {
    curl.push(`-X '${options.method}'`)
  }

if (chain) {
curl.push(
...chain.map(
(item, i) =>
`-H '${AUTH_CHAIN_HEADER_PREFIX + i}: ${JSON.stringify(item)}'`
)
)
}
if (timestamp) {
curl.push(`-H '${AUTH_TIMESTAMP_HEADER}: ${JSON.stringify(timestamp)}'`)
}
if (timestamp) {
curl.push(`-H '${AUTH_METADATA_HEADER}: ${JSON.stringify(metadata)}'`)
}

if (curl.length === 0) {
return null
}

return (

<Grid.Row>
  <Grid.Column mobile="16">
    <Source language="bash" code={curl.join(' \\\n ')} />
  </Grid.Column>
</Grid.Row>
) }

export function AuthChainTable({ chain, timestamp, metadata }) {
  if (!chain) {
    return null
  }

return <Grid.Row>

<Grid.Column mobile="16">
<Grid stackable className="AuthChainTable">
{chain &&
chain.map((item, i) => (<Grid.Row key={AUTH_CHAIN_HEADER_PREFIX + i}>
<Grid.Column mobile="4"><Source code={AUTH_CHAIN_HEADER_PREFIX + i}/></Grid.Column>
<Grid.Column mobile="12"><Source language="json" code={JSON.stringify(item)}/></Grid.Column>
</Grid.Row>))}
{timestamp && <Grid.Row>
<Grid.Column mobile="4"><Source code={AUTH_TIMESTAMP_HEADER}/></Grid.Column>
<Grid.Column mobile="12"><Source language="json" code={JSON.stringify(timestamp)}/></Grid.Column>
</Grid.Row>}
{metadata && <Grid.Row>
<Grid.Column mobile="4"><Source code={AUTH_METADATA_HEADER}/></Grid.Column>
<Grid.Column mobile="12"><Source language="json" code={JSON.stringify(metadata)}/></Grid.Column>
</Grid.Row>}
</Grid>
</Grid.Column>
</Grid.Row>
}

export const TabOptions = {
  HEADERS: 'headers',
  JSON: 'json',
  CURL: 'curl',
}

export function Chain({ chain, timestamp, metadata, options }) {
  const [activeTab, setActiveTab] = useState(TabOptions.HEADERS)

if (!chain) {
return null
}

return (

<Card style={{ width: '100%'}}>
<Card.Content>
<Tabs>
<Tabs.Tab active={activeTab === TabOptions.HEADERS} onClick={() => setActiveTab(TabOptions.HEADERS)}>HEADERS</Tabs.Tab>
{options && <Tabs.Tab active={activeTab === TabOptions.CURL} onClick={() => setActiveTab(TabOptions.CURL)}>CURL</Tabs.Tab>}
<Tabs.Tab active={activeTab === TabOptions.JSON} onClick={() => setActiveTab(TabOptions.JSON)}>JSON</Tabs.Tab>
</Tabs>

        {activeTab === TabOptions.JSON && <Source
          language="ts"
          code={JSON.stringify(chain || null, null, 2)}
        />}

        {options && activeTab === TabOptions.CURL && <AuthChainCURL
          options={options}
          chain={chain}
          timestamp={timestamp}
          metadata={metadata}
        />}

        {activeTab === TabOptions.HEADERS && <AuthChainTable
          chain={chain}
          timestamp={timestamp}
          metadata={metadata}
        />}
          </Card.Content>
          </Card>

)

}

## Auth Chain Generator

export function AuthChainStandar() {
  const [account] = useAuth()
  const [state, setState] = useState({
    method: 'GET',
    url: 'https://events.decentraland.org/api/events',
    metadata: {},
  })
  const [value, setValue] = useState({})
  const [signing, sign] = useAsyncTask(async () => {
    const identity = getCurrentIdentity()
    let url
    try {
      url = new URL(state.url)
    } catch (err) {
      console.error(err)
      return
    }
    if (identity && url) {
      const timestamp = Date.now()
      const chain = await signPayload(
        getCurrentIdentity(),
        [
          state.method,
          url.pathname,
          timestamp,
          (state.metadata && JSON.stringify(state.metadata) || '{}').toLowerCase(),
        ]
          .join(':')
          .toLowerCase()
      )
      setValue({ ...state, timestamp, chain })
    }
  }, [value])
  return (
    <Container>
      <Grid>
        <Grid.Row>
          <Grid.Column tablet="16">
            <ArgsTable
              compact={true}
              updateArgs={(value) => {
                setState((current) => ({ ...current, ...value }))
              }}
              rows={{
                method: {
                  name: 'method',
                  options: [
                    'GET',
                    'POST',
                    'PUT',
                    'PATCH',
                    'DELETE',
                    'HEAD',
                    'CONNECT',
                    'OPTIONS',
                    'TRACE',
                  ],
                  control: {
                    type: 'select',
                    value: state.method,
                  },
                },
                url: {
                  name: 'url',
                  control: { type: 'text', value: state.url },
                },
                metadata: {
                  name: 'metadata',
                  control: {
                    type: 'object',
                    value: state.metadata,
                    placeholder: {},
                  },
                },
              }}
            />
          </Grid.Column>
          <Grid.Column tablet="12">
            {!account && <MissingAuthMessage />}
          </Grid.Column>
          <Grid.Column tablet="4" textAlign="right">
            <Button
              primary
              size="small"
              fluid
              disabled={!account}
              loading={signing}
              onClick={() => sign()}
            >
              Sign
            </Button>
          </Grid.Column>
        </Grid.Row>
        <Grid.Row>
          <Grid.Column tablet="16" >
        <Chain
          options={value}
          chain={value.chain}
          timestamp={value.timestamp}
          metadata={value.metadata}
          />
          </Grid.Column>
        </Grid.Row>
      </Grid>
    </Container>

)
}

## Auth Chain Raw Generator

> Generate an auth chain usign a raw payload

export function AuthChainRaw() {
  const [account] = useAuth()
  const [chain, setChain] = useState(null)
  const [message, setMessage] = useState('get:/path/to/resource')
  const [signing, sign] = useAsyncTask(
    async (payload) => {
      const identity = getCurrentIdentity()
      if (identity) {
        const chain = await signPayload(
          getCurrentIdentity(),
          payload.toLowerCase()
        )
        setChain(chain)
      }
    },
    [chain]
  )
  return (
    <Container>
      <Grid>
        <Grid.Row>
          <Grid.Column tablet="16">
            <ArgsTable
              compact={true}
              updateArgs={(value) => {
                setState(value.message)
              }}
              rows={{
                message: {
                  name: 'message',
                  control: {
                    type: 'text',
                    value: message,
                  },
                },
              }}
            />
          </Grid.Column>
          <Grid.Column tablet="12">
            {!account && <MissingAuthMessage />}
          </Grid.Column>
          <Grid.Column tablet="4" textAlign="right">
            <Button
              primary
              size="small"
              fluid
              disabled={!account}
              loading={signing}
              onClick={() => sign(message)}
            >
              Sign
            </Button>
          </Grid.Column>
        </Grid.Row>
        <Grid.Row>
          <Grid.Column tablet="16">
            <Chain chain={chain} />
          </Grid.Column>
        </Grid.Row>
      </Grid>
    </Container>
  )
}
